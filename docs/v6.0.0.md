#### **[ ✡ 常用方法 ]**

```csharp
Ext.Array.clean(array) : Array // 清空数组
Ext.Array.clone(array) : Array // 浅拷贝
Ext.Array.toArray(iterable, [start], [end]): Array

Ext.Array.insert(array, index, items) : Array 
Ext.Array.remove(array, item) : Array 
Ext.Array.removeAt(array, index, [count]) : Array 

/// <summary>
/// [ 遍历数组 ]
/// Ext.Array.each(array, fn, [scope], [reverse])
/// - fn: function (item, index)
/// - scope: function 函数内 ( this: 作用域  )
/// - reverse: false ( 是否反向遍历 )
/// 
/// </summary>
Ext.Array.each([1, 2, 3, 4, 5], function (item, index) {
    console.log(item);
}, window, true);

const items = Ext.Array.intersect([1, 2, 3], [1, 2]);  # [ 交集 ]
const items = Ext.Array.union([1, 2, 3], [1, 2]);      # [ 并集 ] 去除重复元素
const items = Ext.Array.difference([1, 2, 3], [1, 2]); # [ 差集 ]

Ext.Array.max(array, function(x, y): int); // [ 最大值 ] gt = 1; eq = 0; lt = -1
Ext.Array.min(array, function(x, y): int); // [ 最小值 ] 
Ext.Array.sum(array) : Number  


```
