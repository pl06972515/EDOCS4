#### **[ ✡ 常用方法 ]**

```csharp
# [ 日期操作 ]
/// <summary>
/// - interval: Ext.Date.Year | Ext.Date.MONTH | Ext.Date.DAY
///             Ext.Date.HOUR | Ext.Date.MINUTE | Ext.Date.SECOND | Ext.Date.MILLI( 毫秒 )
///             
///   var now = new Date();
///   console.log(Ext.Date.add(now, Ext.Date.DAY, 5))
/// 
/// </summary>
v.add(date, interval, value) : Date 
v.subtract(date, interval, value) : Date 

/// <summary>
/// - [ 时间差 ]
///   var startDate = new Date('2023-01-01')
///   var endDate = new Date('2023-12-31')
///   
///   console.log('相差年数:', Ext.Date.diff(startDate, endDate, 'y'))
///   console.log('相差月数:', Ext.Date.diff(startDate, endDate, 'mo'))
///   console.log('相差天数:', Ext.Date.diff(startDate, endDate, 'd'))
///   console.log('相差小时数:', Ext.Date.diff(startDate, endDate, 'h'))
///   console.log('相差分数:', Ext.Date.diff(startDate, endDate, 'mi'))
///   console.log('相差秒数:', Ext.Date.diff(startDate, endDate, 's'))
/// 
/// </summary>
v.diff(min, max, unit) : Number 
   
# [ 日期格式化 ] Ext.Date.format(new Date(), 'Y-m-d H:i:s'))
Ext.Date.format(date, format) : String 
    
/// <summary>
/// [ 范围检测 ] 判断 date 是否在 startDate .. endDate 之间    
/// var startDate = new Date('2023-01-01');
/// var endDate = new Date('2023-12-31');
/// var testDate = new Date('2023-06-15');
///
/// var isBetween = Ext.Date.between(testDate, startDate, endDate)
/// console.log(isBetween)  // true
/// 
/// </summary>
Ext.Date.between(date, startDate, endDate) : bool

# [ 其他方法 ]
Ext.Date.isLeapYear(date): bool // 检查日期是否为闰年
Ext.Date.isWeekend(date): bool  // [ 检测 ] 指定日期是否为周末

Ext.Date.getDayOfYear(date): Number    // 获取一年中的第几天
Ext.Date.getDaysInMonth(): Number      // 获取本月总天数
Ext.Date.getFirstDateOfMonth(): Number // 获取本月第一天
Ext.Date.getLastDateOfMonth(): Number  // 获取本月最后一天

Ext.Date.localToUtc(date): Date // [ 日期转换 ] 将本地日期转换 UTC
Ext.Date.utcToLocal(date): Date // [ 日期转换 ] 将 UTC 转换本地日期


```
