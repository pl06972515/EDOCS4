#### **[ ✡ 常用方法 ]**

```csharp
/// <summary>
/// [ 遍历对象 ]
/// Ext.Object.each(object, fn, [scope]) 
/// - fn: function (key, value)
/// - scope: function 函数内 ( this: 作用域  )
/// 
/// </summary>
Ext.Object.each({
    name: 'Jacky',
    hairColor: 'black',
    loves: ['food', 'sleeping', 'wife']
}, function(key, value,) {
    
    console.log(key + ":" + value);
});

/// <summary>
/// [ 查询转换 ] 
/// - 将查询字符串 [ 解析成一个 object ]
///   Ext.Object.fromQueryString(queryString) : Object
/// 
/// - 将 object 转换成 [ 查询字符串 ]
///   Ext.Object.toQueryString(object) : String  
///
/// </summary>
const obj = Ext.Object.toQueryString("x=1&y=2&y=2&z=3");                   // { x: '1', y: ['2', '2'], z: '3' }
const query = Ext.Object.toQueryString({ x: '1', y: ['2', '2'], z: '3' }); // x=1&y=2&y=2&z=3


```
