!> <span style='color:Blue'>[ 在`TS`中，支持泛型`T`]</span>

```csharp
# [ 泛型接口 ]
interface IClass<T> // interface IClass<T extends number>
{
    x: number;
    y: T;
}

# [ 泛型类型 ]
class TClass<T> { 

    private x: T;
    public constructor(x: T) {
        this.x = x;
    }

}

# [ 泛型方法 ]
function M1<T extends number, R extends number>(x: T, y: R): void { 
     
}
```
